appsettings.json içine API URL'sini ekle:

{
  "ApiBaseUrl": "https://localhost:5001/api"
}

builder.Services.AddScoped(sp => new HttpClient { BaseAddress = new Uri("https://localhost:5001/api/") });

veya 
builder.Services.AddHttpClient("API", client =>
{
    client.BaseAddress = new Uri("https://localhost:5001/api/");
});


public class ResourceService
{
    private readonly HttpClient _httpClient;

    public ResourceService(HttpClient httpClient)
    {
        _httpClient = httpClient;
    }

    public async Task<List<ResourceDto>> GetResourcesAsync()
    {
        return await _httpClient.GetFromJsonAsync<List<ResourceDto>>("resources");
    }
}


builder.Services.AddScoped<ResourceService>();


@inject ResourceService ResourceService

@code {
    private List<ResourceDto> resources = new();

    protected override async Task OnInitializedAsync()
    {
        resources = await ResourceService.GetResourcesAsync();
    }
}
